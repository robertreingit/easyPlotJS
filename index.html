A simple javascript charting library knowing just enough to get by.

Single chart

        
var sgraph = eplot.chart()
  .init('#basic')
  .plot(data0);

Connected charts

            
var lgraph = eplot.chart()
  .width(300)
  .height(300)
  .init('#line')
  .plot(data)
  .xTicks(4);
            
          
            
var bgraph = eplot.chart()
  .width(300)
  .height(300)
  .init('#bar')
  .plot(data_bar)
  .yTicks(5);
